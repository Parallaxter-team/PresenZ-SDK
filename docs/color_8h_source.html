849 inline float NozDeltaLAB(NozLAB lab1, NozLAB lab2)
850 {
851  NozLAB deltaLAB = lab1 - lab2;
852 
853  float c1 = sqrt(lab1.a * lab1.a + lab1.b * lab1.b);
854  float c2 = sqrt(lab2.a * lab2.a + lab2.b * lab2.b);
855  float deltaC = c1 - c2;
856 
857  float deltaH = deltaLAB.a * deltaLAB.a + deltaLAB.b * deltaLAB.b - deltaC * deltaC;
858  deltaH = deltaH < 0 ? 0 : sqrt(deltaH);
859

861  const float kc = 1.0f;
862  const float kh = 1.0f;

865  float sh = 1.0f + 0.15f * c1;

869  float deltaHkhsh = deltaH / (kh * sh);
