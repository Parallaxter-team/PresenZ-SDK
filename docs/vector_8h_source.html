14template <typename T>
15bool almostEqual(T a, T b, T relTol = static_cast<T>(1e-5), T absTol = static_cast<T>(1e-12)) {
16 // Compute the difference
17 T diff = std::fabs(a - b);
18
19 // Absolute comparison for very small numbers
20 if (diff <= absTol)
21 return true;
22
23 // Relative comparison scaled by the largest magnitude
24 return diff <= std::max(std::fabs(a), std::fabs(b)) * relTol;
25}
